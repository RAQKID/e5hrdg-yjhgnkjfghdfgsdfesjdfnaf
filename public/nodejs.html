Node.js code for calling the Raqkid API Endpoint:


const axios = require("axios");

const BASE_URL = "https://raqkidapiendpoint505.onrender.com";
const model = "model";   // Model name (Check the model valid names in the official server.)

async function sendPrompt(userPrompt) {
  const params = {
    model: model,
    key: "Your_KEY",      // Replace with your Raqkid API Key
    prompt: userPrompt
  };

  try {
    const response = await axios.get(`${BASE_URL}/${model}`, { params });
    console.log("API Response:", response.data);
  } catch (error) {
    console.error("Error:", error.message);
  }
}

// usage
sendPrompt("Which model are you?");    // Your Prompts

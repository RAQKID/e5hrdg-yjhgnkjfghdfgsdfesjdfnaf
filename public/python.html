import requests

BASE_URL = "https://raqkidapiendpoint505.onrender.com/"

def send_prompt(user_prompt):
    params = {
        "model": "models",     # Model name (Check the model valid names in the official server.)
        "key": "Your_KEY",   # Replace with your Raqkid API Key
        "prompt": user_prompt
    }

    try:
        response = requests.get(BASE_URL, params=params)
        response.raise_for_status()
        print("API Response:", response.json())
    except requests.exceptions.RequestException as e:
        print("Error:", e)

send_prompt("This is my test prompt with spaces!")  # Your Prompts
